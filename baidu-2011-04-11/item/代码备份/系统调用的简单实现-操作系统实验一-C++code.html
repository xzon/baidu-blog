//系统调用的简单实现-操作系统实验一-C++code

#include<iostream>
using namespace std;
/*
系统调用的简单实现
*/
int main()
{
char* cmd;
cout<<"系统命令调用   0 表示退出\n\n";
while(true)
{
   cout<<"\n   请输入Windows下的系统命令：";
   cmd=new char;
   cin.getline(cmd,100,'\n');
   if(cmd[0]=='0')break;
   system(cmd);//   使用接口函数system()进行系统调用
}
return 0;
}
