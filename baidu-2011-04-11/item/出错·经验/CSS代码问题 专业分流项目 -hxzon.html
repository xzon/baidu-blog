1，ul没法设置高度？
ul {
display: block;
}
ul显示成一条线，内部的li显示在它下方。
去掉display，没有什么变化。
ul {
height: 10px;
}
ul在firefox显示出10px高的线，在ie下变成很大一块，把li包住。

3,链接转表单，添加了一个if (uri !== "") 判断。hxzon。
function linkToForm(linkObject) {
var formObject = document.createElement("form");
document.body.appendChild(formObject);
formObject.setAttribute("method", "post");
//formObject.setAttribute("target", "hidden");
var url = linkObject.href;
var urj = "";
var uri = "";
var i = url.indexOf("?");
if (i == -1) {
   urj = url;
} else {
   urj = url.substring(0, i);
}
formObject.action = urj;
if (i >= 0 && url.length >= i + 1) {
   uri = url.substring(i + 1, url.length);
}
if (uri !== "") {
   var sa = uri.split("&");
   for (var j = 0; j < sa.length; j = j + 1) {
    var jsa = sa[j].split("=");
    var inputObject = document.createElement("input");
    inputObject.setAttribute("type", "hidden");
    inputObject.setAttribute("name", jsa[0]);
    inputObject.setAttribute("value", jsa[1]);
    formObject.appendChild(inputObject);
   }
}
formObject.submit();
document.body.removeChild(formObject);
return false;
}
