eclipse,c连接查询mysql -hxzon -ccode


1,用libmysql.lib生成libmysql.a

在Eclipse下用CDT和MinGW编写C访问Mysql时会出现 undefined reference to `mysql_init@4′的问题，这是因为mingw编译时找不到静态库的原因，网上的解决办法一般是用reimp.exe及dlltool.exe重新生成libmysql.def和libmysql.a，然后用libmysql.a。
具体命令是：
reimp -d libmysql.lib
dlltool --input-def libmySQL.def --dllname libmySQL.dll --output-lib libmysql.a -k
//reimp,dlltool在mingw/bin下，3.4.5版本。
//\mysql-5.1.40-win32\lib\opt下有libmysql.lib。

3，C实现
/*
* MysqlAndC.cpp
*
* Created on: 2009-11-16
*      Author: hxzon
*/
#include <windows.h>
#include <stdio.h>
#include <string.h>

#include <mysql.h>

#include "MysqlAndC.h"

int main() {

MYSQL *mysql;
MYSQL_RES *result;
MYSQL_ROW row;
MYSQL *connection;
int qt;
// mysql = (MYSQL *) malloc(sizeof(MYSQL));
// mysql_init(mysql);

mysql = mysql_init(NULL);//初始化，注意用返回值赋值。
connection = mysql_real_connect(mysql, "localhost", "root", "root",
    "analyser_new", 0, NULL, 0);//设置连接参数
if (connection == NULL) {
   printf(mysql_error(mysql));
   return 1;
}

qt = mysql_query(connection, "select * from sys_param");//执行操作

if (qt != 0) {//检查操作的影响
   printf(mysql_error(connection));
   return 1;
}

result = mysql_store_result(connection);//存储操作结果

while ((row = mysql_fetch_row(result)) != NULL) {//循环取得数据行
   printf("int: %s, char: %s\n", row[0], row[1]);
}

mysql_close(connection);//关闭连接
mysql_free_result(result);//释放操作结果
getchar();
return 0;
}
