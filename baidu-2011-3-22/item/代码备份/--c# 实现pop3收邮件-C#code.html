   {
    string decode = "";
    string st = code+"000";//
    string    strcode = st.Substring(0,(st.Length/4)*4);
    byte[] bytes = Convert.FromBase64String(strcode);
    try
    {
     decode = System.Text.Encoding.GetEncoding("GB2312").GetString(bytes);
    }
    catch
    {
     decode = code;
    }
    return decode;
   }
