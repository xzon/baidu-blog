import java.awt.*;
import javax.swing.*;

public class xZComponent{
        public Component self;
       
        public xZComponent(){        }
        public xZComponent(Component self){
                this.self=self;
        }
       
        public xZComponent at(int x,int y,int width,int height){
                self.setBounds(x,y,width,height);
                return this;
        }
       
        public xZComponent in(Container parent){
                parent.setLayout(null);
                parent.add(self);
                return this;
        }
       
        public xZComponent center(xZComponent cp, double hRatio, double vRatio) {
                double s1 = (1-2.0* hRatio)/ (1.0 - hRatio);
                double s2 = (1-2.0*vRatio)/ (1.0-vRatio);
                xZComponent u =new xZAbove( new xZEmpty(),new xZAbove(cp,new xZEmpty(), s2), vRatio);
                return new xZBeside(new xZEmpty(),new xZBeside(u,new xZEmpty(), s1), hRatio);
        }

    private xZComponent hSeq(xZComponent[] cps,int index,int num) {
                if(num == 1) return cps[index];
                return new xZBeside(cps[index], hSeq(cps, index+1, num-1), 1.0/num);
        }
       
    public xZComponent hSeq(xZComponent[] cps){
                return hSeq(cps,0,cps.length);
        }

    private xZComponent vSeq(xZComponent[] cps,int index,int num) {
                if(num == 1) return cps[index];
                return new xZAbove(cps[index], vSeq(cps, index+1, num-1), 1.0/num);
        }
       
    public xZComponent vSeq(xZComponent[] cps){
                return vSeq(cps,0,cps.length);
        }

        public xZComponent block (xZComponent[] cps, int N, int M) {
                xZComponent[][] fcps =formalize(cps, N, M);
                xZComponent[] rows = new xZComponent[fcps.length];
                for(int i = 0; i < fcps.length; i++) {
                        rows[i] = hSeq(fcps[i]);
                }
                return vSeq(rows);
        }
       
        private xZComponent[][] formalize(xZComponent[] cps,int N,int M){
                xZComponent[][] fcps=new xZComponent[N][M];
                for(int ni=0,i=0;ni<N;ni++)
                        for(int mi=0;mi<M;mi++,i++){
                                if( i <cps.length){
                                        fcps[ni][mi]=cps[ i];
                                }
                                else fcps[ni][mi]=new xZEmpty();
                        }
                return fcps;
        }

        public xZComponent blockWithMargin(xZComponent[] cps, int N, int M,
                double hRatio, double vRatio) {
                xZComponent[] ncps = new xZComponent[cps.length];
                for(int i=0; i<cps.length; i++) {
                        ncps[i] = center(cps[i], hRatio, vRatio);
                }
                return block(ncps, N, M);
        }

}

class xZEmpty extends xZComponent {
        public xZComponent at(int x,int y,int width,int height) {
                return this;
        }
        public xZComponent in(Container parent){
                return this;
        }
}
////////////////////////////
class xZAbove extends xZComponent {
        private xZComponent up,low;
        private double upRatio;
       
        public xZAbove(xZComponent up, xZComponent low, double upRatio){
                this.up = up;
                this.low = low;
                this.upRatio = upRatio;
        }
       
        public xZComponent at(int x,int y,int width,int height){
                up.at(x, y, width,(int)(height*upRatio) );
                low.at(x,(int)(y+height*upRatio) , width,(int)(height*(1-upRatio)) );
                return this;
        }
        public xZComponent in(Container parent){
                parent.setLayout(null) ;
                up.in(parent );
                low.in(parent );
                return this;
        }
}
////////////////////////////////////
class xZBeside extends xZComponent {
        private xZComponent left,right;
        private double leftRatio;
       
        public xZBeside(xZComponent left,xZComponent right,double leftRatio){
                this.left = left;
                this.right = right;
                this.leftRatio = leftRatio;
        }
       
        public xZComponent at(int x,int y,int width,int height) {
                left.at(x, y,(int)(width*leftRatio), height);
                right.at((int)(x+ width*leftRatio), y,(int)(width*(1-leftRatio)) , height);
                return this;
        }
       
        public xZComponent in(Container parent){
                parent.setLayout(null);
                left.in(parent);
                right.in(parent);
                return this;
        }
}
