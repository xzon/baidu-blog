Web Services作业（1-2）情况分析
1、作业要求：
（1）作业一：完成一个通信录的XML实例文档；
（2）作业二：写出作业一的XML Schema文档
2、作业中存在的问题
（1）作业一大部分同学完成的比较好，因为简单啊
       存在问题：
       1/ 无XML版本声明；
2/ 无名字空间声明（允许这种情形出现，因为只要求well-form，而不要求validated）
（2）作业二的问题
1/ 缺少根或文档元素（常见的问题）
2/ 命名空间使用不当
例：以下是XML Schema doc:
<xsd:schema xmlns:xsd=" http://www.w3.org/2001/XMLSchema"
            xmlns:apo=" http://www.example.com/PO1 AddressXMLSchema.xsd"
            targetNamespace=" http://www.example.com/PO1"
elementFormDefault="unqualified" attributeFormDefault="unqualified">
<xsd:element name="persons">
      <xsd:complexType>
              <xsd:sequence>
              <xsd:element name="person" type=" personType" maxOccurs="unbounded"/>
              </xsd:sequence>
       </xsd:complexType>
</xsd:element>
以下是关联的 XML Instance doc:
<apo:persons xmlns:apo=" http://www.example.com/PO1"
                   xmlns:xsi=" http://www.w3.org/2001/XMLSchema-instance"
xsi:schemaLocation=" http://www.example.com/PO1 AddressXMLSchema.xsd">
<person personId="001">
           <name personName="Mike">
                     <单位>中国移动</单位>
                     <地址>广州市天河区……***号</地址>
                     <邮政编码>510000</邮政编码>
                   <location> <Tel>020-12345678</Tel></location>
            </name>
   </person>
3/ 类型的定义与引用问题：
<?XML version="1.0"?>
<xsd:Schema XMLns=" http://www.w3.org/2001/XMLSchema"
            targetNameSpace=" http://mywork"/>
<element name=Persons type="person">
     <complexType name="person">
        <sequence>
           <element name="姓名" type="string"/>
           <element name="单位" type="string"/>
           <element name="通信地址" type="string"/>
           <element name="邮政编码" type="number"/>
             <simpleType>
                              <restriction base="string">
                                    <pattern value="\d{6}"/>
                              </restriction>
                        </simpleType>
           <element name="Email" type="string"/>
           <element name="Tel" type="number"/>
<simpleType>
                               <restriction base="string">
                                     <pattern value="\d{4}-d{8}"/>
                               </restriction>
                         </simpleType>
        </sequence>
        <attribute name="id" type="aaa"/>
     </complexType>
</element>
</Schema>
4/ 关于<sequence>、<choice>和<all>
<xsd:complexType name="locationType">
            <xsd:choice>
                <xsd:element name="Tel" type="xsd:string"/>
                <xsd:element name="Email" type="xsd:string"/>
             <xsd:element name="QQ" type="xsd:string"/>
             <xsd:element name="MSN" type="xsd:string"/>
            </xsd:choice>
</xsd:complexType>
5/ XML实例文档中的元素和类型在XML Schema文档中没有定义：注意校验
6/ <simpleType>中<pattern>层面的定义：
<simpleType name="postalcodeType">
   <restriction base="string">
   <pattern value="\d{6}"/>
   </restriction>
</simpleType>
<simpleType name="telType">
   <restriction base="string">
   <pattern value="\d{3}-\d{8}"/>
   </restriction>
</simpleType>
<simpleType name="mobileType">
   <restriction base="string">
   <pattern value="\d{11}"/>
   </restriction>
</simpleType>
<simpleType name="emailType">
   <restriction base="string">
   <pattern value="[A-Za-z]{1}[0-9||A-Za-z||_||.]{3-17}@[0-9||A-Za-z||_]{*}.[a-z]{2-3}"/>
   </restriction>
</simpleType>
