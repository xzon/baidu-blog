private void test2() {
    JPanel jPanel = new JPanel( new MigLayout( "wrap 4" ) ); // 4 cells 換行
    jPanel.add( createTestJLabel("TEST1") );
    jPanel.add( createTestJLabel("TEST2") );
    jPanel.add( createTestJLabel("TEST3TEST3"), "span 2" ); // 當成 2 cells
    jPanel.add( createTestJLabel("TEST4", 50, 50), "span 2 3" ); // 當成寬 2 cells, 高 3 cells
    jPanel.add( createTestJLabel("TEST5") );
    jPanel.add( createTestJLabel("TEST6") );
    jPanel.add( createTestJLabel("TEST7"), "span 2, wrap" ); // 與 wrap 效果一樣, 所以 span 在 wrap 前失效
    jPanel.add( createTestJLabel("TEST5") );
    jPanel.add( createTestJLabel("TEST6") );
    jPanel.add( createTestJLabel("TEST7") );
    show( jPanel );
}
