//功能：计算九连环游戏中取下第n个环需要几步
//整个游戏所需步数=取下第1个环所需步数+...+取下第9个环所需步数
//实现方法：通过递归模拟取环动作
//语言：pascal

program jlh(input,output);
   VAR
     k,s,i,j,total:integer;

   PROCEDURE move(n:integer;VAR s:integer);
     VAR
       j,i:integer;
    BEGIN
     IF (n=1) OR (n=2)
       THEN s:=s+1
       ELSE BEGIN
              FOR i:=1 TO n-2 DO
                move(i,s);
                s:=s+1;
              FOR j:=1 TO n-2 DO
                move(j,s)
            END
    END;

   BEGIN
     k:=0;
     write('input total=');
     read(total);
     move(total,k);
     writeln;
     writeln('s=',k)
   END.
