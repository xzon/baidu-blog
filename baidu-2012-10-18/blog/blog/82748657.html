import org.eclipse.swt.*;
import org.eclipse.swt.custom.*;
import org.eclipse.swt.widgets.*;
public class SashFormExample{
public void run(){
   final Display display=Display.getDefault();
   final Shell shell=new Shell();
   shell.setSize(450,300);
   shell.setText("sashForm example");
  
   final SashForm sashForm=new SashForm(shell,SWT.NONE);
   final Group group1=new Group(sashForm,SWT.BORDER);
  
   final List playList=new List(group1,SWT.SIMPLE);
   playList.setItems(new String[]{
    "tranditional music","pop music"
   });
   playList.setBounds(15,15,120,45);
  
   final Button addButton=new Button(group1,SWT.FLAT);
   addButton.setText("add");
   addButton.setBounds(15,80,100,25);
  
   final List musicList=new List(sashForm,SWT.BORDER);
   musicList.setItems(new String[]{
    "secret garden","two butterfly"
   });
  
   sashForm.setWeights(new int[]{2,3   });
   sashForm.setBounds(10,10,430,400);
  
   shell.layout();
   shell.open();
   while(!shell.isDisposed()){
    if(!display.readAndDispatch()){
     display.sleep();
    }
   }
}//end run

public static void main(String[] args){
   new SashFormExample().run();
}
}
